Chart C

old = filter(final_asthma_df,asthma_age == "asthma_em_5to14")
young = filter(final_asthma_df,asthma_age == "asthma_em_0to4")
fit1 = lm(rate~so2,data = old)
fit2 = lm(rate~so2,data = young)
subplot(
  plot_ly(old, x = ~so2, y = ~rate)%>% 
    add_markers(alpha = 0.5,name = "age from 5 to 14") %>% 
    add_lines(y=fitted(fit1),name = "lm for age 5 to 14"),
  plot_ly(young, x = ~so2, y = ~rate) %>% 
    add_markers(alpha = 0.5,name = "age from 0 to 4") %>% 
   add_lines(y=fitted(fit2),name = "lm for age 0 to 4")
)
